    for j in seq(0, m):
      ------------------- {n : n, m : m, i : i, j : j}
      if i == 0 or j == m - 1:
        ----------------- {n : n, m : m, i : i, j : j}
        dst[i + j] = src[i + j]
        ----------------- {n : n, m : m, i : i, j : j}
      else:
        ----------------- {n : n, m : m, i : i, j : j}
      ------------------- {n : n, m : m, i : i, j : j}
